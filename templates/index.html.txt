    <script>
        // Smooth dynamic player loading matrix without reloading window page
        def_url = "https://www.youtube.com/embed/dQw4w9WgXcQ";
        function loadStream(item) {
            document.getElementById('mainPlayer').src = item.embed_url;
            document.getElementById('activeTitle').innerText = item.title;
            document.getElementById('activeType').innerText = item.type + (item.season ? ` • ${item.season} ${item.episode}` : '');
            document.getElementById('activeQuality').innerText = item.quality;
            document.getElementById('activeLang').innerText = item.language;
            document.getElementById('activeYear').innerText = item.year;
            document.getElementById('activeMetaExtra').innerText = `${item.meta_extra} • Pure 60 FPS streaming engine activated smoothly.`;
            document.getElementById('activeDownload').href = item.download_url;
            
            // Scroll right to player effortlessly
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        // Advanced Search Autocomplete Core Logic
        const searchBox = document.getElementById('searchBox');
        const panel = document.getElementById('suggestionsPanel');

        searchBox.addEventListener('input', async (e) => {
            const query = e.target.value.trim();
            if(query.length < 1) {
                panel.style.display = 'none';
                return;
            }

            try {
                const res = await fetch(`/api/search?q=${encodeURIComponent(query)}`);
                const items = await res.json();
                
                if(items.length === 0) {
                    panel.innerHTML = '<div class="suggestion-item" style="color:var(--text-muted)">No match discovered</div>';
                } else {
                    panel.innerHTML = items.map(item => `
                        <div class="suggestion-item" onclick='triggerSearchSelection(${JSON.stringify(item)})'>
                            <div>
                                <div style="font-size:0.88rem; font-weight:600;">${item.title}</div>
                                <div style="font-size:0.75rem; color:var(--text-muted);">${item.category.toUpperCase()} • ${item.type}</div>
                            </div>
                            <span class="pill quality-pill" style="font-size:0.65rem;">${item.quality}</span>
                        </div>
                    `).join('');
                }
                panel.style.display = 'block';
            } catch (err) {
                console.error("Search index synchronization fail", err);
            }
        });

        function triggerSearchSelection(item) {
            loadStream(item);
            panel.style.display = 'none';
            searchBox.value = '';
        }

        // Clear dropdown panel dynamically when user clicks away
        document.addEventListener('click', (e) => {
            if(!searchBox.contains(e.target) && !panel.contains(e.target)) {
                panel.style.display = 'none';
            }
        });
    </script>
